)ATTR
  ! TYPE(OUTPUT)  INTENS(HIGH) SKIP(ON)
  + TYPE(TEXT)    INTENS(LOW) SKIP(ON)
  % TYPE(TEXT)    INTENS(HIGH) SKIP(ON)
  _ TYPE(INPUT)   INTENS(HIGH) CAPS(ON)
)BODY
%--- BUILDING DEPOSIT LIB/CCF COBOL2 AND DYLAKOR DB2 BATCH COMPILE SERVICES ----
%COMMAND ===>_ZCMD                                                     +
+
% LIBRARIAN MASTER DATASET:
+   PROJECT ===>_LIBPRJ  +
+   GROUP   ===>_LIBGRP  +
+   TYPE    ===>_LIBTYP  +
+   MEMBER  ===>_MEMBER  +                   ACCOUNT CODE:_ACCTCDE         +
+
% SELECT COMPILE TYPE ===>_Z+ (1 THRU 3)
%   1.+ COBOL2 PRE-COMP/COMP/LINK/PACKAGE/BIND  %PACKAGE COLL-ID ===>_DB2COLL +
%   2.+ PACKAGE PROGRAM ONLY                %3.+ BIND PROGRAM ONLY
%      %TURNOVER COMPILE: _Z+  (Y/N)        %WORKORDER NUMBER: _WORKORDR+  WO...
%   4.+ DYLAKOR COMPILE/PRE-COMPILE/ASSEMBLE/LINK/BIND
+
% COBOL2 COMPILE OPTIONS:
+ _DB2OPT                                                       +|
% LOADLIB ===>_CCFLOAD                                     +|
% COPYBOOK AND DCLLIB LIBRARIAN DATASET NODES:    +( select any 4 nodes )
+   ( 1=TESTCOPY  2=TURNCOPY  3=RJCTCOPY  4=QACOPY  5=PRODCOPY  6=EMRGCOPY )
% SYSLIB DD COPYBOOK SEARCH CHAIN:   %1._Z+   %2._Z+   %3._Z+   %4._Z+
% SYSLIB DD DCLLIB SEARCH CHAIN:     %1._Z+   %2._Z+   %3._Z+   %4._Z+
)INIT
  .ZVARS = '(COMPTYP QACOMP C1 C2 C3 C4 D1 D2 D3 D4)'
  .HELP = $CCFDHP1
 IF (&ZERRMSG ¬= ' ') .MSG = &ZERRMSG
  IF (&CURSOR = ' ')
    &CURSOR = MEMBER
    IF (&LIBTYP = ' ')
      &CURSOR = LIBTYP
    IF (&LIBGRP = ' ')
      &CURSOR = LIBGRP
    IF (&LIBPRJ = ' ')
      &CURSOR = LIBPRJ
    .CURSOR = &CURSOR
    &COMPTYP = ' '
)PROC
  VER (&LIBPRJ,DSNAME)
  VER (&LIBGRP,DSNAME)
  VER (&LIBTYP,DSNAME)
  VER (&MEMBER,NAME)
  VER (&ACCTCDE,NONBLANK,PICT,NNNNNNNNNNNNNNNN)
  VER (&COMPTYP,NONBLANK)
  VER (&COMPTYP,RANGE,1,4)
  VER (&QACOMP,LIST,Y,N)
  VER (&CCFLOAD,NONBLANK)
  VER (&C1,NONBLANK)
  VER (&C2,NONBLANK)
  VER (&C3,NONBLANK)
  VER (&C4,NONBLANK)
  VER (&D1,NONBLANK)
  VER (&D2,NONBLANK)
  VER (&D3,NONBLANK)
  VER (&D4,NONBLANK)
  IF (&COMPTYP ¬= 4)
    VER (&DB2COLL,NONBLANK)
  IF (&QACOMP = Y)
    VER (&DB2COLL,NONBLANK)
    VER (&WORKORDR,NONBLANK)
  VPUT (LIBPRJ,LIBGRP,LIBTYP,ACCTCDE) PROFILE
)END
