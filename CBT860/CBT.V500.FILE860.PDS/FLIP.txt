FLIP     TITLE 'F L I P  ***  READ A CARD IMAGE FILE AND FLIP TEXT'
         PUNCH ' ORDER FLIP(P) '   EASIER DEBUG
         COPY  OPTIONGB
         SPACE 1
         SYSPARM LIST=YES
         EJECT ,
         PRINT &PRTSOR
FLIP     PGMHEAD ZERO12,BASE=(R12),PARM=R9,BNDRY=PAGE,                 *
               LOC=BELOW,AM=31,RM=24 (DCB ADDRESSES)
         SPACE 1
         SERVINIT ,
         SERVLOAD @INPREAD,@PRINTER
         PRTOPEN SYSPRINT,OPT=NOWTO
         INPOPEN SYSIN,OPT=ABEND
         LA    R0,4+L'INREC
         STH   R0,OUTRDW
         SPACE 1
GETLOOP  INPGET INREC
         BXH   R15,R15,EXIT
         MVCIN OUTREC,INREC+79
         PRTV  OUTRDW
         B     GETLOOP
         SPACE 1
EXIT     SERVTERM ,
EXITCODE PGMEXIT RC=RETCODE  RETURN WITH CONDITION
         SPACE 1
SYSPRINT PRTWORK SYSPRINT,TITLE=5
SYSIN    INPWORK SYSIN,EODAD=EXIT,EDIT=128
         SPACE 1
SAVE     DSECT ,
DB       DS    D
DB2      DS    D
         SERVDEFS ,          EXPAND SERVICE ADDRESSES
INREC    DS    CL80' '
OUTRDW   DS    AL2(4+80,0)
OUTREC   DS    CL80' '
SAVEEND  EQU   *
         SPACE 1
         IHAPSA ,            LOW
         END   ,
