)ATTR DEFAULT(%+_)
 |    TYPE(OUTPUT)
      INTENS(LOW)
      CAPS(OFF)

 #    TYPE(INPUT)
      INTENS(HIGH)
      PAD(' ')

)BODY EXPAND(//)
%/-/  TSO EXECUTE COMMAND SCREEN  /-/
%COMMAND ===>_ZCMD
%
+Primary Commands:%F+- Function    %S+- Syntax     %O+- Operands
%
+Complete the TSO Command below:
+===>#CMD

+
)MODEL ROWS(SCAN)
    |RECORD
)INIT
 .CURSOR = CMD
 &SAVECMD = &CMD
 &SAVECSR = .CSRPOS
  &ZTDMARK = 'END OF HELP '
  &UMSG40 = &Z
  &LMSG40 = &Z
  &ZTDMSG = DTSM040
  &ZSCROLLD = DATA
)REINIT
 IF (.CURSOR = &Z) .CURSOR = CMD
 IF (.CSRPOS = &Z) .CSRPOS = &SAVECSR
)PROC
 IF (&ZCMD ¬= OPERAND,O,FUNCTION,F,SYNTAX,S,&Z) .MSG=ISPZ001
 IF (&ZCMD ¬= &Z)
     &TYPE = TRUNC (&ZCMD,1)
     IF (&TYPE = S) &TYPE = X
     IF (&TYPE = O) &TYPE = ')'
     &SW = ON
 IF (&CMD = &Z)
   .MSG = DTSM002
   &CMD = &SAVECMD
   REFRESH (CMD)
   .CSRPOS = &SAVECSR
 &UMSG2 = 'NO COMPLETED TSO COMMAND'
 &LMSG2 = 'YOU ARE REQUIRED TO ENTER THE ENTIRE COMMAND OR TO ENTER +
 THE END COMMAND'
)END
