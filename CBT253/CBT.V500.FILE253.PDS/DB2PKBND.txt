/*  REXX EXEC : ( DB2PKBND ).
    FUNCTION  : CALLED BY DB2B BATCH COBOL2 COMPILE TASK FROM SKELETONS
                ( CCFDB2B6 AND CCFDB2B8 ).
                BUILDS THE PACKAGE/BIND/GRANT JOBS FOR SUBMISSION TO
                THE SYSA DSNX SYSTEM.
    INPUT     : &&TSOID.USER.DBRMLIB(&&MEMBER).
    OUTPUT    : SYSS.TESTCCF.ISPFILE.CNTLLIB(&&MEMBER) -
                BATCH JOB SUBMITTED TO SYSA SYSTEM CLASS=P.
                                                                     */
TRACE O
/*      */
HOUSE_KEEPING:
ADDRESS TSO
ARG TSOID ACCTCDE MEMBER DB2COLL
PARMS = 'TSOID ACCTCDE MEMBER DB2COLL'
SRC = 0
DO I = 1 TO 4
   IPARM = WORD(PARMS,I)
   XPARM = VALUE(IPARM)
   IF XPARM ¬= '' THEN ITERATE
   SAY '*** PARM ERROR: MISSING PARM VALUE FOR ('IPARM'). ***'
   SAY '*** DB2PKBND EXEC TASK CANCELLED - RC = (024). ***'
   EXIT 024
END
EXIT 0     /* ***   N O O P   R E T U R N   *** */
GRANTPDS = 'SYSS.TESTCCF.GRANT.CNTLLIB'
ISPFILE = 'SYSS.TESTCCF.ISPFILE.CNTLLIB'
UCNTL = SYSDSN("'"ISPFILE"'")
IF UCNTL = 'DATASET NOT FOUND' THEN DO
   IF SRC ¬= 0 THEN DO
      SAY '*** ALLOCATE ERROR: DATASET NOT FOUND ('ISPFILE'). ***'
      SAY '*** DB2PKBND EXEC TASK CANCELLED - RETURN CODE: ('SRC'). ***'
      EXIT 100
   END
END
X = MSG("OFF")
"FREE FILE(ISPDD)"
"FREE FILE(PARMDD)"

/*     */
MAIN_ROUTINE_DSNX:
DBSUBSYS = 'DSNX'
DB2PLAN = DB2COLL
RUNLIB = 'SYSS.DB2.V2R3M0.DB2X.RUNLIB.LOAD'
DBRMLIB = TSOID||'.USER.DBRMLIB'
CALL BUILD_JOBCARD
CALL BUILD_PACKAGE_IKJEFT01_JCL
CALL BUILD_PACKAGE_SYSIN
CALL CHECK_GRANT_CNTLLIB
INTERPRET CALL LABEL
CALL BUILD_BIND_SYSIN
IF LABEL = 'BUILD_BIND_IKJEFT01_GRANTJCL' THEN CALL BUILD_GRANT_CNTLLIB_JCL
CALL BUILD_IDCAMS_DELETE_MEMBER
CALL WRITE_PKGBIND_JOB
CALL SUBMIT_PKGBIND_JOB
EXIT 0

/*     S U B R O U T I N E S   S E C T I O N     */
/*     */
BUILD_JOBCARD:
"DELSTACK"
JN = MEMBER
JCL.1 = "//"JN" JOB" ACCTCDE",'"TSOID".DB2.PKG.BIND',NOTIFY="TSOID","
JCL.2 = "//          CLASS=P,MSGCLASS=S,MSGLEVEL=(1,1),REGION=6M"
JCL.3 = "/*JOBPARM SYSAFF=SYSA"
DO J = 1 TO 3
   QUEUE JCL.J
END
RETURN

/*     */

JCL.7 = "//SYSIN    DD DUMMY,DCB=BLKSIZE=80"

   QUEUE JCL.J

/*     */
CHECK_GRANT_CNTLLIB:
LABEL = 'BUILD_BIND_IKJEFT01_GRANTJCL'
GMEMB = SYSDSN("'"GRANTPDS"("DB2PLAN")'")
IF GMEMB = 'MEMBER NOT FOUND' THEN RETURN
"ALLOC DA('"GRANTPDS"("DB2PLAN")') F(PARMDD) SHR"
"EXECIO * DISKR PARMDD (FINIS STEM GPARM."
"FREE F(PARMDD)"
DO I = 1 TO GPARM.0
   IF GPARM.I = DBSUBSYS THEN LABEL = 'BUILD_BIND_IKJEFT01_NOGRANT'
END
RETURN

/*     */
WRITE_PKGBIND_JOB:
"ALLOC DA('"ISPFILE"("MEMBER")') FILE(ISPDD) SHR"
SRC = RC
IF SRC ¬= 0 THEN DO
   SAY '*** DATASET ALLOCATE ERROR: ('ISPFILE'('MEMBER'). ***'
   SAY '*** DB2PKBND EXEC SUBTASK CANCELLED - RETURN CODE: ('SRC'). ***'
   EXIT SRC
END
NUMRCDS = QUEUED()
"EXECIO" NUMRCDS "DISKW ISPDD (FINIS"
SRC = RC
"FREE FILE(ISPDD)"
IF SRC ¬= 0 THEN DO
   SAY '*** WRITE ERROR: ERROR WRITING ('ISPFILE'('MEMBER').  ***'
   SAY '*** DB2PKBND EXEC SUBTASK CANCELLED - RETURN CODE: ('SRC'). ***'
   EXIT SRC
END
RETURN

/*     */
SUBMIT_PKGBIND_JOB:

   SAY '*** SUBMIT ERROR: ('OUTPDS'('MEMBER'). ***'
   SAY '*** DB2PKBND EXEC TASK CANCELLED - RETURN CODE: ('SRC'). ***'
   EXIT SRC
END
RETURN
