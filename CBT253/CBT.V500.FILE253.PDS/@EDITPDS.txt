/* REXX EDIT PDS MEMBERS MACRO  */
TRACE O
ADDRESS "ISPEXEC"
"ISREDIT MACRO"
"VGET OLDSTR SHARED"
"VGET NEWSTR SHARED"
IF RC ¬= 0 THEN DO
   SAY '*** MISSING CHANGE FROM STRING OR TO STRING ON COMMAND LINE. ***'
   SAY '*** EDIT PDS MEMBERS MACRO SUBTASK CANCELLED. ***'
   EXIT 12
END
"ISREDIT FIND" OLDSTR
SRC = RC
IF SRC ¬= 0 THEN DO
   "ISREDIT CANCEL"
   EXIT SRC
END
"ISREDIT CHANGE" OLDSTR NEWSTR "ALL"
SRC = RC
"ISREDIT RESET"
"ISREDIT END"
EXIT SRC
