/* REXX EXEC  */
/* LIB/CCF PRODUCTION PROC TURNOVER TO DETERMINE WHEN A PROC WILL BE
   COPIED INTO THE CORRESPONDING JES2 CONCATENATED PDS PROCLIB.  */

TRACE O
/*     */
HOUSE_KEEPING:
ARG PRODMAST
IF PRODMAST = '' THEN DO
   SAY '*** LIBRARIAN MASTER DATASET NAME NOT PASSED AS EXEC ARGMENT. ***'
   SAY '*** CCFPROCP EXEC TASK CANCELLED. ***'
   EXIT 8
END
EXITRC = 4
IF PRODMAST = 'SYS1.PRODPROC.MASTER' THEN EXITRC = 0
EXIT EXITRC
