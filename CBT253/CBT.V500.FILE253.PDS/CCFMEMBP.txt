/* REXX EXEC  */
/* LIB/CCF PRODUCTION PROC TURNOVER TO DETERMINE WHEN A PROC WILL BE
   COPIED INTO THE CORRESPONDING JES2 CONCATENATED PDS PROCLIB.  */

TRACE O
/*     */
HOUSE_KEEPING:
ARG PRODMEMB
IF PRODMEMB = '' THEN DO
   SAY '*** LIBRARIAN MASTER MEMBER NAME NOT PASSED AS EXEC ARGMENT. ** *'
   SAY '*** CCFMEMBP EXEC TASK CANCELLED. ***'
   EXIT 8
END
EXITRC = 4
PDSMEMB = SYSDSN("'SYS1.PRODPROC.MASTER.PDS("PRODMEMB")'")
IF PDSMEMB = 'OK' THEN EXITRC = 0
EXIT EXITRC
