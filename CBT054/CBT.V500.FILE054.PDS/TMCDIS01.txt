)ATTR DEFAULT(%+¢)
  + TYPE(TEXT) INTENS(LOW) SKIP(ON)
  % TYPE(TEXT) INTENS(HIGH) SKIP(ON)
  ¢ TYPE(INPUT) INTENS(HIGH) CAPS(ON) JUST(LEFT)
  { TYPE(OUTPUT) INTENS(LOW) CAPS(ON) JUST(ASIS) PAD(' ')
  ¬ TYPE(INPUT) INTENS(HIGH) CAPS(ON) JUST(ASIS) PAD('_')
)BODY
%---------------------   DELUXE TMC Display for &TMCREG   ----------------------
%COMMAND ===>¢ZCMD                                         %SCROLL ===>¢AMT +
%
%Select codes are:
%    S - Perform Scratch Processing    R - Restore from Scratch Processing
+
%Sel                Data Set Name                  Volser      Scratched by
%--- --------------------------------------------  ------  --------------------
)MODEL CLEAR(TMCNAM)
 ¬Z+{Z                                           +{Z     +{Z                   +
)INIT
 &ZLIST = '(LCMD TMCDSN TMCVOL TMCNAM)'
 .ZVARS = &ZLIST
 .HELP = TMCTUT01
 &AMT = PAGE
 &LCMD = &Z
 &TMCTUT = 'TMCTUT02'
)PROC
  &TMCTUT = 'TMCTUT02'
  IF (&LCMD ¬= R,S,' ')
      &TMCMSGS = 'Invalid selection'
      &TMCMSGL = 'Valid selections are S (scratch) or R (restore)'
      &TMCTUT = 'TMCTUT01'
  VER (&LCMD,LIST,R,S,MSG=TMC000)
)END
