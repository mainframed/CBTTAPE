TITLE SMF 43;
  DATA;
  RETAIN V;
  INFILE DATLIB JFCB=JF MISSOVER;
  IF _N_ = 1 THEN DO;
                  INPUT @;
                  IF INDEX(JF,'SYS1.MAN') ¬=0 THEN V=4; ELSE V=0;
                  END;
  IF V=4 THEN
      INPUT    SMFXXLEN PIB2. SMFXXSEG PIB2. @ ;
  INPUT    SMFXXFLG PIB1. SMFXXRTY PIB1.
                       @;
  IF SMFXXRTY = 212 THEN DO;
                        *PUT '212 DETECTED - SKIPPING';
                         DELETE;
                         END;
  INPUT    TIMEOUT SMFSTAMP8.
      SMFXXSID $CHAR4. @;
  IF SMFXXRTY = 43;
  INPUT SMF43SBS   PIB2. SMF43RSV PIB2.   SMF43LRR PIB2. @ ;
 PUT SMFXXRTY +5 TIMEOUT DATETIME. +5 SMFXXSID +5
 SMF43SBS +5 SMF43RSV +5 SMF43LRR;
  IF SMF43LRR ¬= 0 THEN DO;
  INPUT SMF43RV1 PIB2. SMF43RST PIB1. SMF43OPT PIB1. SMF43EID $CHAR4. ;
    PUT SMF43RV1 +5    SMF43RST +5    SMF43OPT +5    SMF43EID ;
  END;
