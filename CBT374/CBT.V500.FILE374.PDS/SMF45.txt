//SAS     EXEC SAS,OPTIONS=NODATE
//DATLIB  DD DSN=SYS1.MAN1,DISP=SHR
//SYSIN   DD *
TITLE SMF 45;
  DATA;
  RETAIN V;
  INFILE DATLIB JFCB=JF MISSOVER;
  IF _N_ = 1 THEN DO;
                  INPUT @;
                  IF INDEX(JF,'SYS1.MAN') ¬=0 THEN V=4; ELSE V=0;
                  END;
  IF V=4 THEN
      INPUT    SMFXXLEN PIB2. SMFXXSEG PIB2. @ ;
  INPUT    SMFXXFLG PIB1. SMFXXRTY PIB1.
                       @;
  IF SMFXXRTY = 212 THEN DO;
                        *PUT '212 DETECTED - SKIPPING';
                         DELETE;
                         END;
  INPUT    TIMEOUT SMFSTAMP8.
      SMFXXSID $CHAR4. @;
  IF SMFXXRTY = 45;
  INPUT SMF45SBS   PIB2. SMF45RSV PIB2.   SMF45LRR PIB2. @ ;
 PUT SMFXXRTY +5 TIMEOUT DATETIME. +5
 SMFXXSID +5 SMF45SBS +5 SMF45RSV +5 SMF45LRR;
  IF SMF45LRR ¬= 0 THEN DO;
  INPUT SMF45IND PIB2.  SMF45CC PIB2. @ ;
    PUT SMF45IND   +2    SMF45CC
     ;
  END;
