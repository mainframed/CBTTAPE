//C4SYSDMS JOB EDPCSBS1,'SMF239',MSGCLASS=X,CLASS=L,NOTIFY=ED115DM
//*
//SAS     EXEC EDSAS607,OPTIONS='NODATE'
//*ATLIB  DD DSN=SYS1.MAN1,DISP=SHR
//DATLIB  DD DSN=EDH1.PROD.SMFDUMPW.HOLD,DISP=OLD
//SYSIN   DD *
  DATA X239 (KEEP = DATE TI LU US MG TEXT);
  RETAIN V;
  INFILE DATLIB MISSOVER JFCB=JF COL=S LENGTH=LEN;
  IF _N_ = 1 THEN DO;
                  INPUT @;
                  IF INDEX(JF,'SYS1.MAN') ¬=0 THEN V=4; ELSE V=0;
                  END;
  IF V=4 THEN
      INPUT    SMFXXLEN PIB2. SMFXXSEG PIB2. @ ;
  INPUT    SMFXXFLG PIB1. SMFXXRTY PIB1.
                       @;
  IF SMFXXRTY = 212 THEN DO;
                        *PUT '212 DETECTED - SKIPPING';
                         DELETE;
                         END;
  INPUT    TIMEOUT SMFSTAMP8.
      SMFXXSID $CHAR4. @;
  DATE = DATEPART(TIMEOUT);
  IF SMFXXRTY = 239;
  INPUT TI TIME11. @;
  INPUT LU :$CHAR8. @;
  INPUT US :$CHAR8. @;
  INPUT MG :$CHAR8. @;
  RL = LEN -S +1;
  INPUT TEXT $VARYING100. RL;
  IF INDEX(LU,'CEATC') ¬= 0
  OR INDEX(TEXT,'CEATC') ¬= 0;
  RETURN;
  PROC SORT; BY DATE TI;
  PROC PRINT; ID TI;
  FORMAT DATE DATE.;
  FORMAT TI   TIME.;
