         MACRO
&NAME    SPFTBDIS &TBL,&PAN,&MSG,&CUR,&ROW
.*---------------------------------------------------------------------
.*       THIS MACRO WILL INVOKE THE SPF TABLE DISPLAY SERVICE
.*
.*---------------------------------------------------------------------
&NAME    DS    0H
         LCLA  &X
&X       SETA  &SYSNDX
         LCLC  &P,&M,&C,&LIT
         AIF   (T'&CUR EQ 'O').NOCUR
         AIF   ('&CUR'(1,1) NE '''').AAACUR
&LIT     SETC  '&CUR'(2,K'&CUR-2)
         MVC   CURSOR,=CL8'&LIT'
         AGO   .BBBCUR
.AAACUR  ANOP
         AIF   ('&CUR' EQ 'CURSOR').BBBCUR
         MVC   CURSOR,&CUR
.BBBCUR  ANOP
&C       SETC  'CURSOR'
         AIF   (T'&PAN EQ 'O').EEECUR
         AIF   ('&PAN'(1,1) NE '''').CCCCUR
&LIT     SETC  '&PAN'(2,K'&PAN-2)
         MVC   PANEL,=CL8'&LIT'
         AGO   .DDDCUR
.CCCCUR  ANOP
         AIF   ('&PAN' EQ 'PANEL').DDDCUR
         MVC   PANEL,&PAN
.DDDCUR  ANOP
&P       SETC  'PANEL'
.EEECUR  ANOP
         AIF   (T'&MSG EQ 'O').HHHCUR
         AIF   ('&MSG'(1,1) NE '''').FFFCUR
&LIT     SETC  '&MSG'(2,K'&MSG-2)
         MVC   MSG,=CL8'&LIT'
         AGO   .GGGCUR
.FFFCUR  ANOP
         AIF   ('&MSG' EQ 'MSG').GGGCUR
         MVC   MSG,&MSG
.GGGCUR  ANOP
&M       SETC  'MSG'
.HHHCUR  ANOP
         CLI   &C,C' '
         BH    MAAA&X
         CALL  ISPLINK,(TBDISPL,&TBL,&P,&M),VL
         B     MBBB&X
MAAA&X   EQU   *
         CALL  ISPLINK,(TBDISPL,&TBL,&P,&M,&C,&ROW),VL
MBBB&X   EQU   *
         AGO   .XXX
.*------------------------------------------------------------------
.NOCUR   ANOP
         AIF   (T'&MSG EQ 'O').NOMSG
         AIF   ('&MSG'(1,1) NE '''').AAAMSG
&LIT     SETC  '&MSG'(2,K'&MSG-2)
         MVC   MSG,=CL8'&LIT'
         AGO   .BBBMSG
.AAAMSG  ANOP
         AIF   ('&MSG' EQ 'MSG').BBBMSG
         MVC   MSG,&MSG
.BBBMSG  ANOP
&M       SETC  'MSG'
         AIF   (T'&PAN EQ 'O').EEEMSG
         AIF   ('&PAN'(1,1) NE '''').CCCMSG
&LIT     SETC  '&PAN'(2,K'&PAN-2)
         MVC   PANEL,=CL8'&LIT'
         AGO   .DDDMSG
.CCCMSG  ANOP
         AIF   ('&PAN' EQ 'PANEL').DDDMSG
         MVC   PANEL,&PAN
.DDDMSG  ANOP
&P       SETC  'PANEL'
.EEEMSG  ANOP
         CALL  ISPLINK,(TBDISPL,&TBL,&P,&M),VL
         AGO   .XXX
.*------------------------------------------------------------------
.NOMSG   ANOP
         AIF   (T'&PAN EQ 'O').NOPAN
         AIF   ('&PAN'(1,1) NE '''').AAAPAN
&LIT     SETC  '&PAN'(2,K'&PAN-2)
         MVC   PANEL,=CL8'&LIT'
         AGO   .BBBPAN
.AAAPAN  ANOP
         AIF   ('&PAN' EQ 'PANEL').BBBPAN
         MVC   PANEL,&PAN
.BBBPAN  ANOP
&P       SETC  'PANEL'
         CALL  ISPLINK,(TBDISPL,&TBL,&P),VL
         AGO   .XXX
.NOPAN   ANOP
         CALL  ISPLINK,(TBDISPL,&TBL),VL
         AGO   .XXX
.XXX     ANOP
