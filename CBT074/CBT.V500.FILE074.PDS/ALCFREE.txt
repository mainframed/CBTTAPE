         MACRO
&NAME    ALCFREE &FREE=
.*---------------------------------------------------------------------
.*       THIS MACRO IS CALLED BY THE 'SPFALLOC' MACRO TO PROCESS THE
.*       'FREE' KEYWORD
.*
.*       FREE='CLOSE'           IGNORED IF OTHER THAN 'CLOSE'
.*       FREE=NAME              NAME  DC  CL5'CLOSE'
.*
.*---------------------------------------------------------------------
         LCLA  &X
         LCLC  &LIT
&X       SETA  &SYSNDX
&NAME    DS    0H
*----------------------------------------------------------------------
*        FREE=CLOSE
*----------------------------------------------------------------------
         AIF   ('&FREE'(1,1) EQ '''').AAA
         CLC   &FREE,=C'CLOSE'    FREE=CLOSE ?
         BNE   MAAA&X             NO - OK
         AGO   .BBB
.AAA     ANOP
         AIF   ('&FREE'(2,5) NE 'CLOSE').CCC
.BBB     ANOP
         STCM  R14,15,S99TUPTR    TEXT UNIT POINTER
         LA    R1,DALCLOSE        KEY FOR FREE=CLOSE
         STCM  R1,3,S99TUKEY      PUT IN TEXT UNIT KEY FIELD
         MVC   S99TUNUM,=H'0'     MOVE 0 TO TXT UNIT NUMBR FLD
         LA    R14,4(R14)         BUMP TO NEXT TXT UNIT
         LA    R15,4(R15)         BUMP TO NEXT TXT UNIT PTR
MAAA&X   EQU   *
.CCC     ANOP
