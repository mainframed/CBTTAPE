/* rexx */
/* Edit z/OS Unix user command history file                    */
/* from TSO or z/OS Unix.                                      */

/* rexx is placed in a MVS library concatenated to SYSPROC     */
/* or SYSEXEC.                                                 */

/* Examples:                                                   */
/*  Command ===> tso history                                   */

/* In z/OS Unix,                                               */
/*                                                             */
/*  Command ===> tso omvs                                      */
/*  $ tso history                                              */
/*  $ exit                                                     */
/*_____________________________________________________________*/

oedit ".sh_history"
exit
