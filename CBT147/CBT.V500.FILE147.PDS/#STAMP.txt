         MACRO
&LBL     #STAMP &BRANCH=Y,&PTF=NONE,&TRACE=NO
         GBLB  &DEBUG
         LCLC  &A
&A       SETC  '&TRACE'
&DEBUG   SETB  ('&A' EQ 'YES' OR '&A' EQ 'Y')
&A       SETC  '&LBL'
         AIF   ('&A' EQ '').NOLBL
&LBL     DS    0H
.NOLBL   ANOP
         AIF   ('&BRANCH' EQ 'N').A
         B     PASSIT              BRANCH AROUND
         DC    AL1(40)             ID LENGTH
.A       ANOP
         DC    CL8'&SYSECT'        CSECT NAME
         DC    CL8'BRC0610'        FMID
         DC    CL8'&PTF'           RMID (IF ANY)
         DC    CL8'&SYSDATE'       ASSEMBLY DATE
         DC    CL8'&SYSTIME'       ASSEMBLY TIME
         DC    C' ARCHIVER V06.1.3--- COPYRIGHT 1999, '       @RP200703
         DC    C' RICHARD A. FOCHTMAN, JR., WOODRIDGE, IL '
         AIF   ('&BRANCH' EQ 'N').B
         DC    H'4096'             FOR BUMPING BASE REGISTERS
PASSIT   DS    0H
.B       ANOP
         MEND
