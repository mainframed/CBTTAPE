 A = 1
 "EXECIO * DISKR FI001 (STEM READVR."
   DO I = 9 TO READVR.0 - 2
         WORK   = SUBSTR(READVR.I,3,9)
         LHF    = INDEX(WORK,' ')
         LHM    = SUBSTR(WORK,1,LHF - 1)
           LINE.A =  ' EDIT ''SYS1.PDS.CNV('LHM')'' DATA NONUM'
            "EXECIO * DISKW FI002 (STEM LINE."
           LINE.A = COMMAND
            "EXECIO * DISKW FI002 (STEM LINE."
           LINE.A = "     END SAVE"
            "EXECIO * DISKW FI002 STEM LINE."
   END
