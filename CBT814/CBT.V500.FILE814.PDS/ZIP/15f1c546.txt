The CLIST
PROC 0
CONTROL NOMSG NOFLUSH
/* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
/*  clist called by logon proc IKJ42                                 */
/* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * */
    SET NDSNAME=&SYSUID..ISPF42.ISPPROF
    SET ODSNAME=&SYSUID..ISPF.ISPPROF
    FREE FILE(ISPPROF)
    ALLOC FILE (ISPPROF) DSN('&NDSNAME') SHR
    IF &LASTCC = 0 THEN +
      DO
      FREE FILE(ISPCRTE)
      CONTROL MSG
      ATTRIB ISPCRTE DSORG(PO) RECFM(F B) LRECL(80) BLKSIZE(3120)
      ALLOC FILE(ISPPROF) DSN('&NDSNAME') SP(1 1) TRACKS DIR(20) +
            USING(ISPCRTE)
      ALLOC FI(I1) DA('&ODSNAME') SHR REUSE
      ALLOC FI(SYSIN) DA('SYS2.IEBCOPY.SYSIN') SHR REUSE
      CALL 'SYS1.LINKLIB(IEBCOPY)'
      IF &LASTCC = 0 THEN +
        WRITE *** ISPF PROFILE DATASET '&DSNAME' HAS BEEN CREATED ***
      ELSE +
        DO
        FREE FILE(ISPCRTE)
        WRITE
        WRITE *** ERROR: UNABLE TO ALLOCATE ISPF PROFILE DATASET ***
        WRITE
        EXIT CODE(12)
        END
      FREE FILE(ISPCRTE)
      END
    ISPF NOLOGO
    EXIT CODE(0)
