/* BUILDING DEPOSIT, INC. SMS ACS STORGRP ROUTINE                   */
PROC &STORGRP
FILTLIST &VALID_ARTHUR INCLUDE(PARTH.DATA%*.DBA)
FILTLIST &VALID_SYSTEM INCLUDE(SYS%)
FILTLIST &VALID_FDRAUTH INCLUDE(FDR.PASSWORD.V%%%%%%,FDRABR.V*.**)
FILTLIST &DB2_PROD INCLUDE('DB2A','DB2B')
FILTLIST &DB2_TEST INCLUDE('DB2X','DB2Y')
FILTLIST &ESA43M   INCLUDE('SYST','SYSZ')
FILTLIST &PERFCLAS INCLUDE('FASTREAD','FASTWRIT')
IF (&STORCLAS = 'SCTEMP') THEN
    SET &STORGRP = 'USRPOOL','VIOPOOL'
ELSE
     IF (&STORCLAS = &PERFCLAS && &ANYVOL Â¬= '') THEN
         SET &STORGRP = 'PERFPOOL'
ELSE
     IF (&DSN = &VALID_FDRAUTH) THEN
SET &STORGRP = 'USRPOOL','SYSPOOL','DB2PPOOL','CICSPOOL','ARTHPOOL', +
               'DB2TPOOL','PERFPOOL'
ELSE
     IF (&STORCLAS = 'SYSSHARE') THEN
         SET &STORGRP = 'SYSPOOL'
ELSE
     IF (&HLQ = &DB2_PROD) THEN
         SET &STORGRP = 'DB2PPOOL'
ELSE
     IF (&HLQ = &DB2_TEST) THEN
         SET &STORGRP = 'DB2TPOOL'
ELSE
     IF (&DSN = &VALID_ARTHUR) THEN
         SET &STORGRP = 'ARTHPOOL'
ELSE
     IF (&STORCLAS = 'SMSTEST') THEN
         SET &STORGRP = 'SMSTEST'
ELSE
     IF (&MGMTCLAS = 'MVSMAINT') THEN
         SET &STORGRP = 'MNTPOOL'
ELSE
     IF (&HLQ = &ESA43M) THEN
         SET &STORGRP = 'SYSMSYS'
ELSE
     IF (&STORCLAS = 'CICSR' OR &STORCLAS = 'CICSW') THEN
         SET &STORGRP = 'CICSPOOL'
ELSE
     IF (&HLQ = &VALID_SYSTEM) THEN
         SET &STORGRP = 'SYSPOOL'
ELSE
             SET &STORGRP =
                   'USRPOOL'
EXIT CODE(0)
END
