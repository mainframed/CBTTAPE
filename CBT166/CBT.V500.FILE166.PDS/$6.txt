/*                    REXX                                 */

Arg MYLOAD

If MYLOAD = "" Then Do
   Say "DATA SET NAME REQUIRED"
   Exit
   End

MSG = SYSDSN(MYLOAD)

If SYSDSN(MYLOAD) ¬= "OK" Then Do
   Say MSG
   Exit
   End

TEMP = LISTDSI(MYLOAD)

If SYSDSORG  ¬= "PO" & SYSDSORG ¬= "POU" Then Do
   Say "INVALID DSORG FOR LOADLIB AS" SYSDSORG
   Exit
   End

If SYSRECFM ¬= "U" Then Do
   Say "INVALID RECFM FOR LOADLIB AS" SYSRECFM
   Exit
   End

"ISPEXEC VPUT (MYLOAD)"
"ISPEXEC LIBDEF ISPPLIB"
"ISPEXEC LIBDEF ISPLLIB"

"ISPEXEC LIBDEF ISPLLIB DATASET ID("MYLOAD")"
"ISPEXEC LIBDEF ISPPLIB DATASET ID('$VNV.CBT.PDS')"   /** Change **/

"ISPEXEC SELECT PGM(ISRPTC)"

"ISPEXEC LIBDEF ISPPLIB"
"ISPEXEC LIBDEF ISPLLIB"
