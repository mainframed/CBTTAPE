         MACRO
         EDCOMM  &TYPE=
         AIF     ('&TYPE' EQ 'CSECT').CSECT
         AIF     ('&TYPE' EQ 'DSECT').DSECT
         MNOTE   12,'INVALID -TYPE-'
         MEXIT
.CSECT   ANOP
EDCOMM   CSECT
         AGO   .COMM
.DSECT   ANOP
EDCOMM   DSECT
.COMM    ANOP
