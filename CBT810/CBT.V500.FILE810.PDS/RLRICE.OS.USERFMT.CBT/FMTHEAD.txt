* ------------------------------------------------------------------- *
*                                                                     *
*  MODULE NAME: FMTHEAD                                               *
*                                                                     *
*  Process heading (constant) fields during a 'format'                *
*                                                                     *
* ------------------------------------------------------------------- *
FMTHEAD  CSECT
FMTHEAD  AMODE 31
FMTHEAD  RMODE ANY
         USING OSCOMM,R12
         USING SESSION,R11
         USING FMTHEAD,R15
         B     INIT0000
MODID    DC    CL8'FMTHEAD'
         DC    CL8'&SYSDATE'
         DC    CL7'&SYSTIME'
INIT0000 DS    0H
         STM   R14,R12,12(R13)               SAVE REGS
         LR    R10,R15                       COPY ENTRY POINT
         DROP  R15
         USING FMTHEAD,R10                   DEFINE BASE
         L     R9,SESS_DXD_ADDR              WORK AREA
         LR    R15,R9                        COPY WORK AREA ADDRESS
         USING FMT_COMMON,R9
         L     R15,SESS_DXD_ADDR             WORK AREA BASE
         A     R15,DXD_START                 PLUS DISP
         ST    R13,4(R15)                    SAVE R13
         ST    R15,8(R13)                    CHAIN SAVE AREA
         LR    R13,R15                       COPY WORK AREA ADDRESS
         USING DXDFHEAD,R13                  DEFINE WORK AREA BASE
         MVC   DXD_CSECT,MODID
         L     R7,DXD_FMAT_FMAT              FMAT BEING PROCESSED
         USING FMAT,R7
         ITRACE ID=ENTRY,                                              +
               RDATA1=R7
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
         L     R8,COMM_OSSPFD
         USING OSSPFD,R8
         LA    R0,FMAT_DATA                  SOURCE
         LA    R1,L'FMAT_DATA                LENGTH
         LH    R2,FMAT_ROW                   ROW ON DISPLAY
         BCTR  R2,0                          RELATIVE TO ZERO
         MH    R2,SPF_VWIDTH+2               MULTIPLY BY WIDTH
         AH    R2,FMAT_COLUMN                PLUS COLUMN
         BCTR  R2,0                          RELATIVE TO ZERO
         A     R2,DXD_VDATA_ADDR             PLUS VDATA ADDRESS
         ITRACE ID=STRT_END,                                           +
               RDATA1=R0,                                              +
               RDATA2=R2
         LR    R3,R1                         COPY LENGTH
         MVCL  R2,R0                         COPY SOURCE TO DISPLAY
         B     EXIT0000
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
EXIT0000 DS    0H
         ITRACE ID=EXIT
         LH    R15,DXD_RC                    SET RC
         L     R13,4(,R13)                   RESTORE R13
         L     R14,12(,R13)                  RESTORE R14
         LM    R0,R12,20(R13)                RESTORE REGISTERS
         BR    R14                           RETURN
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
DXD_START      DC      Q(DXDFHEAD)

               LTORG
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
DXDFHEAD             DSECT
                     COPY     DXDPREF

DXD_RC               DS   H

DXD_SOURCE_START     DS   A
DXD_SOURCE_END       DS   A
DXD_DISPLAY_START    DS   A
DXD_DISPLAY_END      DS   A
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
               COMMON
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
               SESSION
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
               BPXYSTAT DSECT=YES,LIST=YES
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
FMT_COMMON     DSECT
               DS      (DXD_PREFIX_L)X
               COPY    FMTCOMM
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
               COPY    FMAT
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
               OSSPFD  TYPE=DSECT
* ------------------------------------------------------------------- *
*                                                                     *
* ------------------------------------------------------------------- *
               COPY    TRENTRY
* ------------------------------------------------------------------- *
*              EQUATES                                                *
* ------------------------------------------------------------------- *
               COPY    REGEQU
               END     FMTHEAD
