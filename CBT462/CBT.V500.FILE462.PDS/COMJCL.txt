  ISPEXEC CONTROL ERRORS RETURN

  IF &STR(&PARM.) = &STR(HELP) THEN +
    DO
       ISPEXEC DISPLAY PANEL(#COMJCL)
     EXIT CODE(0)
    END

      ISPEXEC SETMSG MSG(ISRZ001)

    END

  ISREDIT (DEST) = LINENUM .ZDEST
