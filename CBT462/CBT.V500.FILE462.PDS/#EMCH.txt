%TUTORIAL ------------------ EDIT PRIMARY COMMANDS -------------------- TUTORIAL
%OPTION  ===>_ZCMD                                                             +
+
%                     -----------------------------------
                      |    Installation Edit Macros     |
                      -----------------------------------
+
 The following topics are presented in sequence, or may be selected by
 entering the%EDIT MACRO+name.
+
+Edit Macros are:
+
+   COMMENT   -    ONLY      -    PROFLIST     -    GENJCL
+   TSOTOJCL  -    SHOWDCB   -    APPEND       -    BOX
+   CONVISPF  -    CORNER    -    ECHANGE      -    EXEC
+   CENTER    -    LJUST     -    CUT          -    PASTE
+   NEWMSGS   -    ACFCOMP   -    COMASM       -    COMCLIST
+   COMEXEC   -    COMEXED   -    COMJCL       -    COMPDS
+   CONLY     -    DEBLANK
+
)PROC
   &ZSEL = TRANS(&ZCMD
           COMMENT,#COMMENT
              ONLY,#ONLY
          PROFLIST,#PROFLIS
            GENJCL,#GENJCL
          TSOTOJCL,#TSOTOJC
           SHOWDCB,#SHOWDCB
            APPEND,#APPEND
               BOX,#BOX
          CONVISPF,#CONVISP
            CORNER,#CORNER
           ECHANGE,#ECHANGE
              EXEC,#EXEC
            CENTER,#CENTER
             LJUST,#LJUST
               CUT,#CUT
             PASTE,#PASTE
           NEWMSGS,#NEWMSGS
           ACFCOMP,#ACFCOMP
            COMASM,#COMASM
          COMCLIST,#COMCLIS
           COMEXEC,#COMEXEC
           COMEXED,#COMEXED
            COMJCL,#COMJCL
            COMPDS,#COMPDS
             CONLY,#CONLY
           DEBLANK,#DEBLANK
                )
   &ZUP = ISR20000
)END
