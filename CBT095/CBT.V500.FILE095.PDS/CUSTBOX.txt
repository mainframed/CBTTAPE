ISREDIT MACRO (WDTH,DPTH)
      IF &WDTH = ? THEN DO
      ISPEXEC DISPLAY PANEL(CUSTBOX)
      EXIT
      END

ISREDIT (ROW,COL) = CURSOR
ISPEXEC CONTROL ERRORS RETURN
IF &WDTH = THEN SET &WDTH = 20
IF &DPTH = THEN SET &DPTH = 10
SET &DASHES = &STR(----------------------------------------)
SET &DASHES = &STR(&DASHES&DASHES)
SET &OBEN = &STR(+&SUBSTR(1:&WDTH-2,&DASHES)+)
SET &UNTEN = &STR(+&SUBSTR(1:&WDTH-2,&DASHES)+)
SET &MAX = &DPTH - 2
SET &CNT = 1
ISREDIT LINE &ROW               = LINE + < &COL '&OBEN'>
DO WHILE(&CNT LE &MAX)
   ISREDIT LINE &EVAL(&ROW+&CNT) = +
     LINE + < &COL '|' &EVAL(&COL+&WDTH-1) '|'>
   SET &CNT = &CNT + 1
END
ISREDIT LINE &EVAL(&ROW+&CNT) = LINE + < &COL '&UNTEN'>
IF &MAXCC > 0 THEN DO
      SET ZEDSMSG = &STR(INCOMPLETE BOX)
           SET ZEDLMSG = &STR(NOT ENOUGH ROOM TO DRAW A COMPLETE BOX)
   ISPEXEC SETMSG MSG(ISRZ000)
END
SET &COL = &COL + 1
SET &ROW = &ROW + 1
ISREDIT CURSOR = (ROW,COL)
EXIT CODE(0)
