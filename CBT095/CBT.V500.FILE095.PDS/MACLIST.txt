      IF &WNW = ? THEN DO
      ISPEXEC DISPLAY PANEL(MACLIST)
      EXIT
      END

WRITE UPDATE MODE ABORTED

GOTO NOWRITE
END

GOTO PROCESS
END

      ISPEXEC TBOPEN EDITMACS &WNW
    END
NEXT: +

ISPEXEC TBDISPL EDITMACS PANEL(MACLIST) MSG(&MSG)
