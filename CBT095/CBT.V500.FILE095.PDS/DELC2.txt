 ISREDIT MACRO (PARM) NOPROCESS
 ISPEXEC CONTROL ERRORS RETURN
                      ISREDIT (ROW,COL)     = CURSOR
                      IF &ROW NE 1 THEN +
                      ISREDIT (LAST)    = LINENUM .ZLAST
                      IF &ROW > &LAST THEN +
                      SET &ROW = &ROW + 1
 SET LEVEL = 1
 START: +
ISPEXEC VGET (RCUTLM&LEVEL) PROFILE
  IF &LASTCC = 0 THEN DO
  SET &CUT = YES
  SET &FLAG = YES
  END
  ELSE  DO
  SET &CUT = NO
  END
SET LINES = &&RCUTLM&LEVEL
ISPEXEC VERASE RCUTLM&LEVEL PROFILE
  ISPEXEC VERASE (RCUTLB&LEVEL,RCUTRB&LEVEL) PROFILE
 DO WHILE &INDEX < &EVAL(&LINES + 1)
 ISPEXEC VERASE RCUT&LEVEL&INDEX  PROFILE
   SET INDEX  = &INDEX + 1
 END
 CNTUP: +
 IF &CUT = YES THEN DO
 SET &N1 = &STR(*LEVEL &LEVEL WITH &EVAL(&INDEX - 1) LINES DELETED)
 ISREDIT LINE_BEFORE &ROW = MSGLINE "&N1"
 END

 SET &N1 = &STR(NO CUT LINES FOUND IN PROFILE TO DELETE)

 END
