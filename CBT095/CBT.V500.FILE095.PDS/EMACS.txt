ISREDIT MACRO (WNW,DEBUG)
      IF &WNW = ? THEN DO
      ISPEXEC DISPLAY PANEL(EMACS)
      EXIT
      END
IF &DEBUG = DEBUG  THEN +
CONTROL LIST SYMLIST CONLIST
IF &WNW ¬= THEN DO
WRITENR TABLE UPDATE LIMITED TO MACRO ADMINISTRATOR ENTER PASSWORD:
READ &PASS
IF &PASS ¬= ADMINISTRATOR THEN DO
WRITE UPDATE MODE ABORTED
SET &WNW = NOWRITE
GOTO NOWRITE
END
SET &WNW = WRITE
GOTO PROCESS
END
NOWRITE: +
SET &WNW = NOWRITE
PROCESS: +
SET &MSG =&STR()
 ISPEXEC CONTROL ERRORS RETURN
ISPEXEC TBOPEN EDITMACS &WNW
  SET &RCC = &LASTCC
  IF &RCC = 8 THEN  +
    DO
      WRITE THE MASTER TABLE DOES NOT EXIST WE WILL MAKE ONE
      SET &WNW= WRITE
      SET &MACNAME = B
      SET &MACDESC = BROWSE MEMBER OF CURRENT DATA
      ISPEXEC TBCREATE EDITMACS NAMES(MACNAME MACDESC) WRITE
      ISPEXEC TBADD EDITMACS
      ISPEXEC TBOPEN EDITMACS &WNW
    END
NEXT: +
    ISPEXEC TBSORT EDITMACS FIELDS(MACNAME)
AFTER: +
ISPEXEC TBDISPL EDITMACS PANEL(MACLIST) MSG(&MSG)
IF &LASTCC = 8 THEN SET &EXIT = YES
IF &OPTN = A THEN DO
SET &OPTN = &ZCMD
END
IF &WNW = WRITE THEN DO
IF &OPTN=A && &WNW=WRITE THEN DO
    SET &MACNAME =
    SET &MACDESC =
    ISPEXEC DISPLAY PANEL(MACADD2)
    ISPEXEC TBADD EDITMACS
    END
IF &OPTN=D && &WNW=WRITE THEN +
    ISPEXEC TBDELETE EDITMACS
 END
   ISPEXEC TBCLOSE EDITMACS
   IF &EXIT = YES THEN EXIT
ISPEXEC TBOPEN EDITMACS &WNW
SET &OPTN =
    IF &SUBSTR(1:1,&STR(&ZCMD)) = &STR(L) THEN DO
        SET &MACNAME = &STR()
        IF &LENGTH(&STR(&ZCMD)) > 3 THEN DO
            IF &SUBSTR(3:3,&STR(&ZCMD)) ¬= $STR( ) THEN DO
                SET &MACNAME = &SUBSTR(3:&LENGTH(&STR(&ZCMD)),&STR(&ZCMD))
                SET &ZCMD  = &STR()
                ISPEXEC TBSARG EDITMACS NEXT NAMECOND(MACNAME,EQ)
                END
                END
                END
      ISPEXEC TBSCAN EDITMACS ARGLIST(MACNAME)
      IF &LASTCC = 0 THEN GOTO AFTER
   GOTO NEXT
            END
