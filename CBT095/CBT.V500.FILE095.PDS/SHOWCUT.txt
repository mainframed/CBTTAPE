 ISPEXEC CONTROL ERRORS RETURN
 IF &PARM Â¬= &Z THEN +
   DO
     IF &PARM = ? THEN +
       DO
         ISPEXEC DISPLAY PANEL(SHOWCUT)
         EXIT
       END
     ELSE +
        IF &DATATYPE(&PARM) = NUM THEN +
          DO
           SET LEVEL = &PARM
           END
        ELSE +
          DO
           SET ZEDSMSG = &STR(INVALID ARGUMENT)
             SET ZEDLMSG = &STR(VALID ARGUMENTS ARE A "?" OR A LEVEL +

           ISPEXEC SETMSG MSG(ISRZ001)
          EXIT
        END
  END

IF &LASTCC > 0 THEN +
  DO
           SET ZEDSMSG = &STR(LEVEL DOES NOT EXIST)
             SET ZEDLMSG = &STR(NO CUT LINES FOR LEVEL &LEVEL EXIST)
           ISPEXEC SETMSG MSG(ISRZ001)
           EXIT CODE(12)
        END
SET LINES = &&RCUTLM&LEVEL

WRITE CUT LEVEL &LEVEL HAS &LINES LINES
 DO WHILE &INDEX < &LINESTOPASTE
 ISPEXEC VGET RCUT&LEVEL&EVAL(&INDEX+1) PROFILE
 SET &CUT = &STR(&&RCUT&LEVEL&EVAL(&INDEX+1))
 SET &LEN = &LENGTH(&STR(&CUT))
 IF &LEN > 79 THEN SET &LEN = 79
 IF &LENGTH(&STR(&CUT)) NE 0 THEN +
   SET &CUTX  = &SUBSTR(1:&LEN,&STR(&CUT))

 IF &LENGTH(&STR(&CUT)) = 0 THEN  SET &CUTX = &STR(*** BLANK LINE ***)
WRITE &CUTX
   SET INDEX  = &INDEX + 1
 END
