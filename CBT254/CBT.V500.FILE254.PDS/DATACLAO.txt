/* BUILDING DEPOSIT, INC. SMS ACS DATACLASS ROUTINE                 */
PROC &DATACLAS
/* DASD UNITS                                                       */
FILTLIST DASD_UNITS INCLUDE(
    '3380','3390','DISK','PAGEIO','SYSDA','SYSSQ','SYSTEST',
    'SMS','QA','PARTH',
    'VIO','SYSALLDA','APPLTA','SEQDA','SYSTA','USRDA','')
/* MGMTCLASS AND DSOWNER VALUES THAT ARE CHECKED HERE               */
FILTLIST OWNERS INCLUDE(
    'FASHNBUG','CORPSVC1','CORPSVC2','STORESYS')
/* IGNORE IF NON-DASD ALLOCATION AND SET DATACLASS TO "NONSMS"      */
IF &UNIT Â¬= &DASD_UNITS THEN DO
     SET &DATACLAS = 'NONSMS'
     EXIT CODE(0)
     END
/* IF THE USER CODED A MGMTCLASS THEN PROCESS BASED ON THAT VALUE   */
/* IF THE USER CODED NULL THEN USE THE DSOWNER VALUE FOR PROCESSING */
SELECT
/* SET VALUE TO ACTUAL USERID OF ALLOCATOR                          */
     WHEN (&MGMTCLAS = &OWNERS | +
           (&MGMTCLAS = '' && +
             (&DSOWNER = &OWNERS))) DO
       SET &DATACLAS = '$USER'
       EXIT CODE(0)
       END
/* SET VALUE TO SPACE ACCOUNTING GROUP IF MGMTCLASS=TEMP            */
     WHEN (&MGMTCLAS = 'TEMP') DO
       SET &DATACLAS = '$DSOWNER'
       EXIT CODE(0)
       END
/* ELSE SET A DEFAULT VALUE OF SMSUSER                              */
     OTHERWISE DO
       SET &DATACLAS = 'SMSUSER'
       EXIT CODE(0)
       END
END
END
