/* REXX COMPARE XA TSO HLQ LIST TO ESA TSO HLQ LIST */
TRUE="1";FALSE="0";
"ALLOC F(INXA) DA('TECH110.USER.CNTLLIB(TIDOUT)') SHR";
"ALLOC F(INESA) DA('TECH110.USER.CNTLLIB(TIDOUTE)') SHR";
"ALLOC F(OUTPUT) DA('TECH110.USER.CNTLLIB(TIDXCEPT)') SHR";
"EXECIO * DISKR INXA (FINIS STEM IXA.";
"EXECIO * DISKR INESA (FINIS STEM IESA.";
DO I =1 TO IXA.0;
   IXA.I=STRIP(IXA.I);
END;
DO I =1 TO IESA.0;
   IESA.I=STRIP(IESA.I);
END;
INXANESA="";
INESANXA="";
OUTCTR=1;
XAPTR=1;
ESAPTR=1;
XAEOF=FALSE;
ESAEOF=FALSE;
DO WHILE((¬XAEOF) & (¬ESAEOF));
    SELECT;
      WHEN (IXA.XAPTR=IESA.ESAPTR) THEN DO;
             XAPTR=XAPTR+1;ESAPTR=ESAPTR+1;
             IF XAPTR>IXA.0 THEN XAEOF=TRUE;
             IF ESAPTR>IESA.0 THEN ESAEOF=TRUE;
             END;
      WHEN (IXA.XAPTR>IESA.ESAPTR) THEN DO;
             INESANXA=INESANXA IESA.ESAPTR;
             ESAPTR=ESAPTR+1;
             IF ESAPTR>IESA.0 THEN ESAEOF=TRUE;
             END;
      WHEN (IXA.XAPTR<IESA.ESAPTR) THEN DO;
             INXANESA=INXANESA IXA.XAPTR;
             XAPTR=XAPTR+1;
             IF XAPTR>IXA.0 THEN XAEOF=TRUE;
             END;
    END;
END;
DO WHILE(¬XAEOF);
             INXANESA=INXANESA IXA.XAPTR;
             XAPTR=XAPTR+1;
             IF XAPTR>IXA.0 THEN XAEOF=TRUE;
END;
DO WHILE(¬ESAEOF);
             INESANXA=INESANXA IESA.ESAPTR;
             ESAPTR=ESAPTR+1;
             IF ESAPTR>IESA.0 THEN ESAEOF=TRUE;
END;
OUT.OUTCTR="TSO HLQ IN XA AND NOT IN ESA";
OUTCTR=OUTCTR+1;
OUT.OUTCTR=" ";
OUTCTR=OUTCTR+1;
HII=WORDS(INXANESA);
DO I = 1 TO HII;
   OUT.OUTCTR = WORD(INXANESA,I);
   OUTCTR=OUTCTR+1;
END;
OUT.OUTCTR=" ";
OUTCTR=OUTCTR+1;
OUT.OUTCTR="TSO HLQ IN ESA AND NOT IN XA";
OUTCTR=OUTCTR+1;
OUT.OUTCTR=" ";
OUTCTR=OUTCTR+1;
HII=WORDS(INESANXA);
DO I = 1 TO HII;
   OUT.OUTCTR = WORD(INESANXA,I);
   OUTCTR=OUTCTR+1;
END;
ADDRESS "TSO" "EXECIO * DISKW OUTPUT (FINIS STEM OUT.";
EXIT 0;
