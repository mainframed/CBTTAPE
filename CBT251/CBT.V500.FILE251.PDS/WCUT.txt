 /********************* WCUT     ISPF EDIT MACRO *********************/
 /* WCUT IS A DIFFERENT KIND OF CUT MACRO.  WCUT (WINDOW CUT) ALLOWS */
 /* YOU TO CUT A SMALL  PART  OF  A  DATASET  AND PASTE THE CUT PART */
 /* SOMEWHERE ELSE USING THE WPASTE  MACRO.   A WORD, LINE, GROUP OF */
 /* CHARACTERS, OR ANY WINDOW OF YOUR DISPLAYED SCREEN CAN BE CUT BY */
 /* WCUT.  ENTER WCUT ON THE  COMMAND  LINE  AND PLACE THE CURSOR ON */
 /* THE FIRST CHARACTER TO  BE  INCLUDED.   PRESS ENTER.  THEN PLACE */
 /* THE CURSOR AT THE END OF THE  AREA OR WINDOW TO BE CUT AND PRESS */
 /* PF12.  THE AREA IN THE  WINDOW  DEFINED BY CURSOR PLACEMENT WILL */
 /* BE CUT AND CAN BE  LATER  PASTED  WITH WPASTE.  AS LITTLE AS ONE */
 /* CHARACTER CAN BE CUT AND PASTED.                                 */
 /*                                                                  */
 /* SYNTAX: WCUT                                                     */
 /*                                                                  */
 /********************************************************************/

  ISREDIT MACRO (HELP)

       IF &STR(&HELP) = HELP THEN +
           DO
           ISPEXEC DISPLAY PANEL(WCUT)
           EXIT
           END

       IF &STR(&HELP) = ? THEN +
           DO
           ISPEXEC DISPLAY PANEL(WCUT)
           EXIT
           END

  CONTROL  ASIS
  ISPEXEC CONTROL ERRORS RETURN
  CONTROL ASIS
  SET LCC = &LASTCC
  ISREDIT (ROW,COL) = CURSOR
  SET &WCROWS = &ROW
  SET &WCCOLS = &COL
  ISPEXEC VPUT (WCCOLS WCROWS) PROFILE
  ISPEXEC VGET (ZPF12) PROFILE
  SET &WCZPF12 = &ZPF12
  SET &ZPF12 = WCUT2
  ISPEXEC VPUT (ZPF12 WCZPF12) PROFILE
