PROC 1 JB
CONTROL MAIN MSG
ISPEXEC CONTROL ERRORS RETURN
GOTO NORM
CONTROL LIST CONLIST SYMLIST
NORM:-
SET &JJ = &JB&STR(*)
SET &JBN = &STR(&JJ)
/* Q &JJ */
/* WRITE */
SET SYSOUTTRAP = 99
CONTROL MSG
STATUS &JB
CONTROL NOMSG
SET &S = SYSOUTLINE
SET &N = 1
DO WHILE 1=1
    SET &L = &STR(&&&S&N)
    IF &STR(&L) = &STR() THEN GOTO DISPL
    IF &SUBSTR(1:9,&L) = &STR(IKJ56211I) THEN -
     DO
     SET &I1 = &EVAL(20+&LENGTH(&JB))
     SET &I2 = &EVAL(23+&LENGTH(&JB))
     SET &JBN = &SUBSTR(&I1:&I2,&L)
     IF &JBN < 1000 THEN -
     SET &JBN = &SUBSTR(&I1+1:&I2,&L)
     IF &JBN < 100 THEN -
     SET &JBN = &SUBSTR(&I1+2:&I2,&L)
     IF &JBN < 10  THEN -
     SET &JBN = &SUBSTR(&I1+3:&I2,&L)
     END
    SET &N = &N +1
END
DISPL:-
SET &ZZ = &STR(TSO F;ACT;S S &JBN)
ISPEXEC DISPLAY PANEL(ZONK) COMMAND(ZZ)
EXIT CODE(0)
