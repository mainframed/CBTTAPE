ISREDIT  MACRO (APARM)
/*  MACRO NAME : BLOCK                                              */
/*    FUNCTION : CREATE MULTIPLE BLOCK CHARACTERS IN TEXT           */
/*               (SUPER-TYPE-TEXT)                                  */
/*      SYNTAX : BLOCK T=STRING TO GENERATE AFTER POINTED LINE      */
/*               BLOCK OR BLOCK T=                                  */
/*                   AND STRING TO GENERATE READ FROM POINTED LINE  */
/*               BLOCK HELP OR H                                    */
/*       DEBUG : USE L= IN PLACE OF T=                              */
/*      ORIGIN : CBT TAPE 89 FEB, FILE 95                           */
/*      UPDATE : SOME CHANGED FROM ORIGINAL (MOINIL P.A.)           */
  SET &AMPSD = &STR(&&)
  SET &TL = &LENGTH(&NRSTR(&APARM))
  IF &TL = 1 AND &NRSTR(&APARM) = &STR(H) OR +
    &TL = 4 AND &NRSTR(&APARM) = &STR(HELP) THEN DO
      ISPEXEC CONTROL DISPLAY SAVE
      ISPEXEC SELECT PGM(ISPTUTOR) PARM(PAJTSTT1)
      SET &RCX = &LASTCC
      ISPEXEC CONTROL DISPLAY RESTORE
      EXIT CODE(&RCX)
      END
  CONTROL MSG
  IF &TL > 2 THEN DO
    SET &TC = &SUBSTR(1:2,&NRSTR(&APARM))
    IF &STR(&TC) = &STR(L=) THEN CONTROL LIST SYMLIST CONLIST
    IF &STR(&TC) = &STR(T=) OR +
      &STR(&TC) = &STR(L=) THEN +
      SET &CHARS = &SUBSTR(3:&TL,&NRSTR(&APARM))
    ELSE DO
      SET ZEDSMSG = INVALID OPERAND
      SET ZEDLMSG = ENTER "BLOCK HELP" COMMAND FOR OPERAND SYNTAX
      ISPEXEC SETMSG MSG(ISRZ000)
      EXIT CODE(0)
      END
    END
  IF &TL < 3 THEN DO
    IF &TL = 2 AND &NRSTR(&APARM) = &NRSTR(L=) THEN +
      CONTROL LIST SYMLIST CONLIST
    IF &NRSTR(&APARM.) = &STR() OR +
      &TL = 2 AND ( &NRSTR(&APARM) = &NRSTR(T=) OR +
      &NRSTR(&APARM) = &NRSTR(L=) ) THEN DO
        ISREDIT (ROW,COL) = CURSOR
        ISPEXEC CONTROL ERRORS RETURN
        ISREDIT (CHARS) = LINE &ROW
        IF &LASTCC = 12 THEN DO
          SET ZEDSMSG = OPERAND MISSING
          SET ZEDLMSG = ENTER "BLOCK HELP" COMMAND FOR OPERAND SYNTAX
          ISPEXEC SETMSG MSG(ISRZ000)
          EXIT CODE(0)
          END
        ISPEXEC CONTROL ERRORS CANCEL
        SET &CHARS = &CHARS
        SET &ROW = &ROW + 1
        END
    ELSE DO
      SET ZEDSMSG = INVALID OPERAND
      SET ZEDLMSG = ENTER "BLOCK HELP" COMMAND FOR OPERAND SYNTAX
      ISPEXEC SETMSG MSG(ISRZ000)
      EXIT CODE(0)
      END
    END
  SET &X = &LENGTH(&NRSTR(&CHARS))
  IF &X = 0 THEN EXIT
  ISREDIT (XLB,XRB) = BOUNDS
  SET &XNCL = (&XRB-&XLB) + 1
  SET &XNC = &XNCL / 6
  SET &XNL = &X / &XNC
  SET &XW = &XNL * &XNC
  IF &XW ¬= &X THEN SET &XNL = &XNL + 1
  SET &XNL = &XNL * 6
  IF &X < &XNC THEN SET &XSC = 1
  ELSE SET &XSC = (((&XNCL - (&XNC*6)) + 1) / 2) + 1
  ISREDIT LOCATE 1
  IF &LASTCC = 8 THEN DO
    ISREDIT NONUM
    SET &XN = 0
    ISREDIT MASKLINE = 'X'
    DO WHILE &XN < &XNL
      ISREDIT LINE_AFTER &XN = MASKLINE
      SET &XN = &XN + 1
      END
    ISREDIT C 'X' ' ' ALL
    END
  ELSE DO
    ISREDIT LOCATE .ZCSR
    ISREDIT MASKLINE = '{'
    ISREDIT LINE_AFTER .ZCSR = MASKLINE
    SET &XN = 1
    DO WHILE &XN < &XNL
      ISREDIT (LINEPTR) = LINENUM .ZCSR
      SET &LINEPTR = &LINEPTR + &XN
      ISREDIT LINE_AFTER &LINEPTR = MASKLINE
      SET &XN = &XN + 1
      END
    ISREDIT C '{' ' ' ALL
    END
  ISREDIT MASKLINE = ' '
  ISREDIT RESET
  ISPEXEC CONTROL ERRORS RETURN
  ISREDIT (ROW,COL) = CURSOR
  SET &ROW = &ROW + 1
  SET ROWX = &ROW
  SET ROWN = &ROW
  ISREDIT (REC) = LINE &ROW
  SET &XN = 0
  SET &XROW = &XNC + 1
  SET &XCOL = 1
LOOP: +

    SET &LTR = &SUBSTR(&XN:&XN,&NRSTR(&CHARS))
    IF &XN = &XROW THEN DO
      SET &ROW = &ROW + 6
      SET &XROW = &XROW + &XNC
      END
    IF &XN = &XCOL THEN DO
      SET &COL = &XSC
      SET &XCOL = &XCOL + &XNC
      END
    ELSE DO
      SET &COL = &COL + 6
      IF &SPA = Y THEN SET &COL = &COL - 3
      END
    SET &SPA =
    END
  IF &XN > &X THEN DO
    IF &X < &XNC THEN DO
      SET &D = &EVAL((&XNCL-(&X*6))+1)
      SET &OVER = &EVAL(&D/2)
      ISREDIT (BOUNDS) 1 &XNCL
      SET &ROW = &ROWX
      ISREDIT NULLS ON
      ISREDIT (CURSOR) = &ROW 1
      ISREDIT SHIFT ) &ROW &OVER
      ISREDIT SHIFT ) &EVAL(&ROW + 1) &OVER
      ISREDIT SHIFT ) &EVAL(&ROW + 2) &OVER
      ISREDIT SHIFT ) &EVAL(&ROW + 3) &OVER
      ISREDIT SHIFT ) &EVAL(&ROW + 4) &OVER
      ISREDIT NULLS OFF
      END
    ISREDIT LOCATE &ROWN
    EXIT
    END
