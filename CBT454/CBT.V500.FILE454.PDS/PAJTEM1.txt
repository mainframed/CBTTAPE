%Tutorial --------------------------  EDIT  --------------------------- Tutorial
%COMMAND ===>_ZCMD                                                             +
+
%                     -----------------------------------
                      |      Edit Macro  "EDMODE"       |
                      -----------------------------------
+
%STATS :+
   Use%STATS+to turn STATS mode on or off ("ON" is the default).
   If STATS mode is on, ISPF statistics will be generated for members of
   partitioned data-sets. If STATS mode is off, no ISPF statistics will be
   generated. STATS mode is always set off for sequential data-sets.

   More topic will be presented in the EDIT tutorial.
+
+Enter%&EPF+to exit the tutorial and return to the EDIT modes control panel
+Hit%ENTER+to go on to the next tutorial panel
)INIT
  &EPF = PFK(END)
  IF (&EPF = &Z)
    &EPF = 'END'
  &D = TRUNC(&EPF,2)
  &DT = .TRAIL
  &D = TRUNC(&DT,1)
  IF (&D = '0')
    &DT = .TRAIL
    &EPF = 'PF&DT'
)PROC
)END
