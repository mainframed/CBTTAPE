%----------------- Generation Data Group Utility -------------------------------
%Command ===>_ZCMD
+
+                  %---+GDG%model+data-set label Utility%---+
+
+GDG%model+data-set label is :%&GDGPREF
+
%Enter/verify parameters below :+
+
+   Volume serial%===>_G0    + (must be the volume on which the%Index+resides,
+                               so, you should first LIST the GDG to know it)
+
%   Only for Create :+- - - - - - - - - - - - - - - - - - - - - - - - - - - - -
+
+           RECFM%===>_G1   %(*)+(left all these parameters blanks for%Delete+
+         BLKSIZE%===>_G2   %(*)+ operation.%Beware+to fill any one which will
+           LRECL%===>_G3   %(*)+ cause a%Create+operation)
+                           %(*)---> Note :+values are%not+verified.
+
+
+Press%ENTER+to proceed or%END+to terminate.
+
)INIT
  .HELP = PAJTGDGM
  .CURSOR = G0
  &G0 = &Z
  &G1 = &Z
  &G2 = &Z
  &G3 = &Z
)PROC
  VER (&G0,NB,NAME)
  IF (&G = S)
    IF (&G1 ¬= &Z)
      VER (&G1,ALPHA)
    IF (&G2 ¬= &Z)
      VER (&G2,NUM)
    IF (&G3 ¬= &Z)
      VER (&G3,NUM)
  IF (&G = T)
    &G1 = &Z
    &G2 = &Z
    &G3 = &Z
  VPUT (G0 G1 G2 G3) SHARED
)END
