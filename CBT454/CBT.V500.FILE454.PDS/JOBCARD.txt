ISREDIT  MACRO (APARM) NOPROCESS
/*  MACRO NAME : JOBCARD                                            */
/*    FUNCTION : COPY JOB CARD FROM USER ISPF PROFILE               */
/*      SYNTAX : JOBCARD                                            */
/*               JOBCARD HELP OR H                                  */
/*       DEBUG : JOBCARD LIST                                       */
/*      ORIGIN : CBT TAPE 89 FEB, FILE 95                           */
  IF &STR(&APARM.) Â¬= &STR() THEN DO
    IF &STR(&APARM.) = &STR(HELP) OR +
      &STR(&APARM.) = &STR(H) THEN DO
        ISPEXEC CONTROL DISPLAY SAVE
        ISPEXEC SELECT PGM(ISPTUTOR) PARM(PAJTJBCD)
        SET &RCX = &LASTCC
        ISPEXEC CONTROL DISPLAY RESTORE
        EXIT CODE(&RCX)
        END
    IF &STR(&APARM.) = &STR(LIST) THEN CONTROL LIST CONLIST SYMLIST
    ELSE DO
      SET ZEDSMSG = INVALID OPERAND
      SET ZEDLMSG = VALID OPERAND ARE HELP OR NULL
      ISPEXEC SETMSG MSG(ISRZ000)
      EXIT CODE(0)
      END
    END
  ISREDIT PROCESS DEST
  IF &LASTCC = 0 THEN DO
    ISREDIT LOCATE .ZDEST
    IF &LASTCC = 8 THEN GOTO EMPTY
    ISPEXEC VGET ZLLGJOB1
    ISREDIT MASKLINE = "&ZLLGJOB1"
    ISREDIT LINE_AFTER .ZDEST = MASKLINE
    ISPEXEC VGET ZLLGJOB2
    ISREDIT (LINEPTR) = LINENUM .ZDEST
    SET &LINEPTR = &LINEPTR + 1
    ISREDIT MASKLINE = "&ZLLGJOB2"
    ISREDIT LINE_AFTER &LINEPTR = MASKLINE
    ISPEXEC VGET ZLLGJOB3
    ISREDIT (LINEPTR) = LINENUM .ZDEST
    SET &LINEPTR = &LINEPTR + 2
    ISREDIT MASKLINE = '&ZLLGJOB3'
    ISREDIT LINE_AFTER &LINEPTR = MASKLINE
    ISPEXEC VGET ZLLGJOB4
    ISREDIT (LINEPTR) = LINENUM .ZDEST
    SET &LINEPTR = &LINEPTR + 3
    ISREDIT MASKLINE = '&ZLLGJOB4'
    ISREDIT LINE_AFTER &LINEPTR = MASKLINE
    GOTO ERROROUT
    END
  ELSE DO
EMPTY: ISPEXEC VGET ZLLGJOB1
    ISREDIT MASKLINE = "&ZLLGJOB1"
    ISREDIT LINE_AFTER 0 = MASKLINE
    ISPEXEC VGET ZLLGJOB2
    ISREDIT (LINEPTR) = LINENUM .ZCSR
    SET &LINEPTR = .ZLAST
    ISREDIT MASKLINE = "&ZLLGJOB2"
    ISREDIT LINE_AFTER 1 = MASKLINE
    ISPEXEC VGET ZLLGJOB3
    ISREDIT (LINEPTR) = LINENUM .ZCSR
    SET &LINEPTR = .ZLAST
    ISREDIT MASKLINE = '&ZLLGJOB3'
    ISREDIT LINE_AFTER 2 = MASKLINE
    ISPEXEC VGET ZLLGJOB4
    ISREDIT (LINEPTR) = LINENUM .ZCSR
    SET &LINEPTR = .ZLAST
    ISREDIT MASKLINE = '&ZLLGJOB4'
    ISREDIT LINE_AFTER 3 = MASKLINE
ERROROUT: ISREDIT MASKLINE = ' '
    SET RC = 0
    END
  END
EXIT CODE(&RC)
