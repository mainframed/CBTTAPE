PROC     0 MOFB
         CONTROL NOMSG
         FREE F(BRODCAST SYSOUT)
         DELETE BRODSCAN.REPORT
         CONTROL MSG
         ISPEXEC VGET (MOFB)
         ALLOC F(BRODCAST) DA('SYS1.BRODCAST') SHR
         IF &MOFB ¬= YES THEN GOTO DIRL
         ALLOC F(SYSOUT) DA(BRODSCAN.REPORT) NEW BL(1580) SPA(46,23)
         GOTO XLST
DIRL:    ALLOC F(SYSOUT) DA(*)
XLST:    CALL '...TARGET.LINKLIB(BRODSCAN)'
         CONTROL NOMSG
         IF &MOFB ¬= YES THEN GOTO FRPL
         ISPEXEC BROWSE DATASET(BRODSCAN.REPORT)
         DELETE BRODSCAN.REPORT
FRPL:    FREE F(BRODCAST SYSOUT)
         FREE DA('...TARGET.LINKLIB')
         CONTROL MSG
