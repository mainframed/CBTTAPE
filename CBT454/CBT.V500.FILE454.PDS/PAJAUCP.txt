+Command%===>_RCMD

@Z

              ==================================='
)PROC
  VER (&RCMD,NONBLANK)
  &CHKC = TRUNC (&RCMD,' ')
  IF (&CHKC = 'F')
    &RCMD = TRUNC (&RCMD,' ')
    &CHCK = .TRAIL
    IF (&CHCK = &Z)
      VER (&FCMD,NONBLANK)

      &FCMD = &CHCK
  &RAUT = TRUNC (&RAUT,2)
  &RAUT = TRANS (&RAUT ON,ON OF,OFF *,*)

)END
