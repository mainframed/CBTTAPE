PROC 1 FNUM PREF(SBGOLOB) OUTPUT(X) NUMB(486) CBT(CBT) FLE(@FILE) -
       DSNM(CBTDOC.PDS)
/* ------------------------------------------------------------- */
/* CLIST TO FIX COLUMNS 73-80 IN THE CBT DOCUMENTATION, FOR A    */
/* SPECIFIC FILE NUMBER.  DATASET IN WHICH THE MEMBER IS FIXED,  */
/* IS THE STAGING DATASET FOR THAT CBT TAPE FILE.                */
/* ------------------------------------------------------------- */
     SET &FILVER = &STR(CBT&NUMB)
     SET &FILNAM = &STR(&FILVER..FILE&FNUM)
     SET &DATAST = &STR('&PREF..&FILNAM')
     SET &VER    = &SUBSTR(1:1,&NUMB)
     SET &MODIF  = &SUBSTR(2:3,&NUMB)
     WRITE ISPF STATS:  VERSION    IS &VER
     WRITE ISPF STATS:  MODIF LVL  IS &MODIF
     ALLOC F(INPUT)  SH REUSE DA('&DATAST(&FLE&FNUM)')
     ALLOC F(OUTPUT) SH REUSE DA('&DATAST(&OUTPUT)')
     WRITE FIXING COLUMNS 73-80 FOR CBTDOC IN MEMBER &FLE&FNUM
     WRITE DATASET BEING FIXED:  &DATAST
     WRITE OUTPUT MEMBER NAME :  &OUTPUT
     CALL 'SBGOLOB.LOAD(DOCFILE)'
     WRITE PDS86 BEING CALLED TO RENAME OUTPUT MEMBER
     PDS86 '&DATAST' XISPM
     ATTRIB &OUTPUT ADDSTATS ID(&CBT-&NUMB) VER(&VER) MOD(&MODIF)
     RENAME &OUTPUT &FLE&FNUM SWAP
     DELETE &OUTPUT
     END
     WRITE MEMBER &FLE&FNUM SUCCESSFULLY RENAMED
     FREE FI(INPUT OUTPUT)
     WRITE INPUT AND OUTPUT FILES FREED
     WRITE DOCFIL EXECUTION COMPLETED WITH RETURN CODE &LASTCC
