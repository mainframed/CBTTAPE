/* REXX **************************************************************/
/* MEMBER : ALOCHELP                                                 */
/* DESCRIPTION:                                                      */
/* Allocate the dataset userid.HELP          or the dataset given as */
/* parm as file SYSHELP.                                             */
/*********************************************************************/
ARG help_dsn
SIGNAL ON ERROR

DDNAME = SYSHELP
BLANK = ' '
DSLIST= ' '
DA = NO
X=OUTTRAP(SYSOUTLINE.)
ADDRESS TSO "LISTALC STATUS"
X=OUTTRAP(OFF)
L = 1
LINES = SYSOUTLINE.0
DO WHILE L <= LINES
   LINE = SYSOUTLINE.L
   ZW = LEFT(LINE,1)
   IF ZW = BLANK THEN
      DO
      DD = SUBSTR(LINE||'             ',3,8)
      DD = DD
      IF DD = DDNAME THEN
         DO
         DA = YES
         LL = L
         END
      END
   L = L + 1
   END
IF help_dsn = '' THEN DO
   help_dsn = Userid()||'.HELP'
END
DSN = "'"help_dsn"'"
TESTDSN = help_dsn
IF DA = NO THEN DO
   SAY ' DD 'DDNAME' NOT ALLOCATED, ALLOCATING 'DSN' TO 'DDNAME
   ADDRESS TSO "ALLOCATE F("DDNAME") DSN("DSN") SHR"
   CC = RC
   IF CC ¬= 0 THEN DO
      SAY 'ALLOCATION FOR 'DSN' FAILED, RC='CC
      EXIT 4
      END
   EXIT 0
   END
I = 0
NDD = NO
DO WHILE NDD = NO
   LL = LL - 1
   I = I + 1
   SAVEDSN.I = SYSOUTLINE.LL
   LL = LL + 3
   IF LL > SYSOUTLINE.0 THEN NDD = YES
   ELSE DO
      LINE = SYSOUTLINE.LL
      LINEDD = SUBSTR(LINE||'             ',3,8)
      LINEDD = LINEDD
      IF LINEDD ¬= BLANK THEN NDD = YES
      END
   END
II = 1
DO WHILE II <= I
   DSNAME = SAVEDSN.II
   IF DSNAME = TESTDSN THEN DO
      EXIT 0
      END
   DSNAME = "'"||DSNAME||"'"
   DSLIST = DSLIST||"      "||DSNAME
   II = II + 1
   END
"FREE F("DDNAME")"
"ALLOC F("DDNAME") DA("DSN" "DSLIST") SHR"
CC = RC
IF CC ¬= 0 THEN DO
   SAY 'ALLOCATION FAILED'
   SAY 'RESTORING OLD ALLOCATION'
   "ALLOC F("DDNAME") DA("DSLIST") SHR REUSE"
   EXIT 4
   END
SAY "SYSHELP '"help_dsn"' ALLOCATED"
EXIT 0

/*********************************************************************/
/* subroutines                                                       */
/*********************************************************************/
ERROR:
 say 'REXX error' rc 'in line' sigl
 say "SOURCELINE"(sigl)
RETURN
