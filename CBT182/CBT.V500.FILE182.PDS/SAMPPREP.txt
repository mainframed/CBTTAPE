    PROC 2 MEMN DSNN  VOL(NONE) DEBUG
    IF &DEBUG=DEBUG THEN CONTROL LIST CONLIST SYMLIST
    ISPEXEC CONTROL ERRORS RETURN
    SET CLR=D
    SET NONDPLY='N'
      ISPEXEC  CONTROL NONDISPL
      ISPEXEC  DISPLAY  PANEL(SAMPPRE1)
      IF &LASTCC ¬= 0 THEN  DO          /* RETURN CODES */
        EXIT
        END
      ISPEXEC  DISPLAY  PANEL(SAMPPRE1)
      IF &LASTCC ¬= 0 THEN  DO          /* RETURN CODES */
        EXIT
        END
    IF &MEM1¬= THEN SET MEM2=&MEM1
    ELSE SET MEM2=&MEMN
    IF &REP=YES THEN SET &REP=REPLACE
    ELSE SET &REP=NOREPL
    IF &VOL = NONE THEN SET INVOL=&Z
    ELSE SET INVOL=&STR(INVOL(&VOL))
    IF &VOL1 = &Z   THEN SET OUTVOL=&Z
    ELSE SET OUTVOL=&STR(OUTVOL(&VOL1))
    ISPEXEC SELECT PGM(ISPPREP) PARM(INPAN('&DSNN(&MEMN)') -
                 OUTPAN('&ENT1(&MEM2)') &INVOL &OUTVOL &REP)
    SET &SAMPRC=&LASTCC
      ISPEXEC  DISPLAY  PANEL(SAMPPRE2)
    EXIT CODE(&SAMPRC)
