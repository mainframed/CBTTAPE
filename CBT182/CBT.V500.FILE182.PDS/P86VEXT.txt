)ATTR DEFAULT(%+_)
    % TYPE(TEXT) INTENS(HIGH) COLOR(WHITE) SKIP(ON)
    ¢ TYPE(TEXT) INTENS(HIGH) COLOR(YELLOW)
    + TYPE(TEXT) INTENS(LOW)  COLOR(TURQ) SKIP(ON)
    _ TYPE(INPUT) INTENS(HIGH) CAPS(ON) JUST(LEFT) COLOR(RED)
    ¬ TYPE(INPUT) INTENS(HIGH) CAPS(ON) JUST(LEFT) COLOR(RED) PAD(_)
    ¦ TYPE(INPUT) INTENS(HIGH) CAPS(OFF)
)BODY EXPAND(\\)
+-\-\-¢PDS Log/Table Output+-\-\-
%COMMAND ===>_ZCMD
+
+
+ Specify output destination:
+      Output dataset  %===>_PDSPXDSN
+      Dataset DISP    %===>_Z+   (%O+is OLD;%N+is NEW;%M+is MOD;%S+is SHR)
+
+
+      SYSOUT Class    %===>_Z+
+      SYSOUT FORM     %===>_Z   +
+      SYSOUT DEST     %===>_Z       +
+      SYSOUT Copies   %===>_Z +       (1 to 99)
+
+
+      Note: A%SYSOUT+class will take precedence over an output data set name.
)INIT
   .HELP  = P86VEXTH
   .ZVARS = '(PDSPDISP PDSPCLAS PDSPFORM PDSPDEST PDSPNUM)'
   &ZCMD = &Z

   IF (&PDSPXDSN = &Z)
      &PDSPXDSN = 'PDSLOG.DATA'
   IF (&PDSPDISP = &Z)
      &PDSPDISP = 'O'
   IF (&PDSPNUM = &Z)
      &PDSPNUM = '1'

   IF (.CURSOR = &Z)
      IF (&PDSPCLAS = &Z)
         .CURSOR = PDSPXDSN
      IF (&PDSPCLAS ¬= &Z)
         .CURSOR = PDSPCLAS

)PROC
   IF (&ZCMD ¬= &Z) .MSG = ISPZ001

   IF (&PDSPCLAS = &Z)
      VER(&PDSPXDSN,NB,DSNAME)
      VER(&PDSPDISP,LIST,NB,O,N,M,S)

   IF (&PDSPCLAS ¬= &Z)
      VER(&PDSPNUM,NB,RANGE,1,99)

   IF (.MSG = &Z)
      &TEMP = TRUNC (&PDSPXDSN,'''')
      &TRAIL = .TRAIL
      IF (&TRAIL = &Z)
         &EXTRDSN = '&ZPREFIX..&PDSPXDSN'
      IF (&TRAIL ¬= &Z)
         &EXTRDSN = TRUNC (&TRAIL,'''')
      &EXTRDSN = TRUNC (&EXTRDSN,'(')
      &MEMBER = .TRAIL
      &MEMBER = TRUNC (&MEMBER,')')

      VPUT(PDSPXDSN PDSPDISP PDSPCLAS PDSPDEST PDSPFORM PDSPNUM) PROFILE
)END
+  ¢Warning:+Do not allocate%NEW+output datasets with a low level qualifier of
+           %LOADxxxx.+ These allocations default to RECFM=U.
