)ATTR DEFAULT(%+_)
    % TYPE(TEXT) INTENS(HIGH) COLOR(WHITE) SKIP(ON)
    ¢ TYPE(TEXT) INTENS(HIGH) COLOR(YELLOW)
    + TYPE(TEXT) INTENS(LOW)  COLOR(TURQ) SKIP(ON)
    _ TYPE(INPUT) INTENS(HIGH) CAPS(ON) JUST(LEFT) COLOR(RED)
)BODY
+        Panel preprocessing target Dataset entry panel
%COMMAND ===>_ZCMD                                                             +
%
%   Preprocessing member &MEMN in dataset &DSNN
%
%   Enter dataset name of target Panel PDS
%
+   PROJECT ===>_PROJECT +
+   LIBRARY ===>_LIBRARY +
+   TYPE    ===>_TYPE    +
+   MEMBER  ===>_MEMBER  +  member if different than source member
+
+OTHER PARTITIONED OR SEQUENTIAL DATA SET:
+   DATA SET NAME  ===>_DSN1
+   VOLUME SERIAL  ===>_VOL1  +  (If not cataloged)
+
+Replace existing member+===>_REP+(Yes or No)                        +
)INIT
 IF (&CLR='D')
   &PROJECT=&ZPREFIX
   &LIBRARY=&LIB0
   &TYPE=&TYP0
   &CLR='C'
 IF (&CLR='C')
   &MEMBER=''
   &DSN1=''
   &VOL1=''
   &CLR=''
 IF (&REP=&Z) &REP=YES
)PROC
   &REP=TRANS(TRUNC(&REP,1),Y,YES,N,NO,*,YES)
   VER (&REP,NB,LIST,YES,NO)
   IF (&DSN1='')
     VER (&PROJECT,NB,NAME)
     VER (&LIBRARY,NB,NAME)
     VER (&TYPE,NB,NAME)
     VER (&MEMBER,NAME)
     &MEM1=&MEMBER
     &ENT1='&PROJECT..&LIBRARY..&TYPE'
   IF (&DSN1 ¬= '')
     VER (&DSN1,DSNAME)
     &ENT1=TRUNC(&DSN1,'(')
     &MEM1=''
     IF (&ENT1 ¬= &DSN1) /* MEMBER NAME GIVEN
       &MEM1=.TRAIL
       &MEM1=TRUNC(&MEM1,')')
       &T=TRUNC(&ENT1,1)
       IF (&T='''')
         &ENT1=.TRAIL
       IF (&T¬='''')
         &ENT1='&ZPREFIX..&ENT1'
     IF (&MEM1='') /* NO MEMBER NAME GIVEN
       &T=TRUNC(&DSN1,1)
       IF (&T='''')
         &ENT1=.TRAIL
         &ENT1=TRUNC(&ENT1,'''')
       IF (&T¬='''')
         &ENT1='&ZPREFIX..&DSN1'
       VER(&MEMBER,NAME)
       &MEM1=&MEMBER
   VER (&MEM1,NAME)
   VER (&VOL1,NAME)
   VER (&ENT1,NB,DSNAME)
)END
)PROC
)END
