 PROC 3 SRC OBJ MEM
 ISPEXEC CONTROL ERRORS RETURN
 SET TITLE=ASSEMBLER PROMPTER
 SET CLR=R
   ISPEXEC  CONTROL NONDISPL
   ISPEXEC  DISPLAY  PANEL(SCPENT1)
   IF &LASTCC ¬= 0 THEN  DO          /* RETURN CODES               */
     EXIT
     END                             /* 12 - PANEL, MESSAGE, OR    */
 DISPR: -
 SET REQ1=&STR(BROWSE OUTPUT Y/BLANK)
 SET ENT1=
 SET REQ2=PRINT ID OR CLEAR
 SET ENT2=&MEM
 SET REQ3=LINK MEMBER OR CLEAR
 SET ENT3=&MEM
 SET REQ12=A REQUEST FOR BROWSE WILL CAUSE REDISPLAY BEFORE PRINT/LINK
   ISPEXEC  DISPLAY  PANEL(SCPENT1)
   IF &LASTCC ¬= 0 THEN  DO          /* RETURN CODES               */
     EXIT
     END                             /* 12 - PANEL, MESSAGE, OR    */
 IF &ENT1=Y | &ENT1=B THEN DO
 ISPEXEC  BROWSE  DATASET(ASM.LIST)
 GOTO DISPR
 END
 IF &ENT2¬= THEN -
 PRINTDS ASM.LIST ID(&ENT2)  BOX('&SRC')
 IF &ENT3 =  THEN EXIT
 SET &MEM2=&ENT3
 SET CLR=R
   ISPEXEC  CONTROL NONDISPL END
   ISPEXEC  DISPLAY  PANEL(SCPDSN)
   IF &LASTCC > 12 THEN  DO          /* RETURN CODES               */
     EXIT
     END                             /* 12 - PANEL, MESSAGE, OR    */
 SET TYPE=LOAD
 SET REQ1=   ENTER LOAD LIBRARY DSN
 SET REQ2=   ENTER LINK OPTIONS
 SET REQ3=   ENTER MORE OPTIONS
 SET ENT2=&STR(NCAL LET REUS RENT REFR)
 SET ENT3=&STR(AC(1) LIST XREF PRINT(*))
   ISPEXEC  DISPLAY  PANEL(SCPDSN)
   IF &LASTCC ¬= 0 THEN  DO          /* RETURN CODES               */
     EXIT
     END                             /* 12 - PANEL, MESSAGE, OR    */
 IF &MEM1= THEN SET &MEM1=&MEM2
 IF &MEM1 = THEN SET &LOAD=&ENT1
 ELSE SET &LOAD=&ENT1(&MEM1)
 LINK '&OBJ'  LOAD('&LOAD') &ENT2 &ENT3
