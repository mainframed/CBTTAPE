*
* CODING EXAMPLE 6.1.4
*
*
* CALLING PROGRAM:
*
*
PROG01   CSECT
         ...
         ESPIE SET,EXIT01,(1,4,7)
         ...
         LINK  EP=PROG02
         ...
         END
*
*
*
*
* CALLED PROGRAM:
*
*
PROG02   CSECT
         ...
SETESPIE ESPIE SET,EXIT02,(11)
         ST    R1,TOKEN            SAVE TOKEN OF PREV ESPIE ENVIR
         BR    R6                  RETURN TO CALLING RTN
         ...
RETURN   ESPIE RESET,TOKEN         RESTORE ESPIE ENVIR OF CALLING PROG
         L     R13,4(0,R13)        RESTORE ADR OF PREV SA
         LM    R14,R12,12(R13)     RESTORE REGS OF CALLING PROG
         BR    R14                 RETURN TO MVS OR CALLING PROG
         ...
TOKEN    DS    F
         ...
         END
