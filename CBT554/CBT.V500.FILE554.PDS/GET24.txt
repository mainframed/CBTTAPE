         LCLB  &E1                                                         3
         LCLC  &LBL1,&LBL2                                                 4
.**********************************************************************    5
.*       PERFORM 24 BIT I/O IN 31 BIT PROGRAM                         *    6
.*       REG = PARAMETER NAMES A WORK REGISTER                        *    7
.**********************************************************************    8
         AIF   (T'&REG NE 'O').OK1                                         9
         MNOTE 8,'OPERAND "REG" IS REQUIRED'                              10
&E1      SETB  1                                                          11
.OK1     ANOP                                                             12
         AIF   (&E1).MEND                                                 13
&LBL1    SETC  'GTT'.'&SYSNDX'                                            14
&LBL2    SETC  'GTX'.'&SYSNDX'                                            15
         LA    &REG,&LBL1                                                 16
         LA    15,&LBL2                                                   17
         BSM   &REG,15                                                    18
&LBL2    DS    0H                                                         19
         GET   &DCB,&AREA,&RPL=                                           20
         BSM   0,&REG                                                     21
&LBL1    DS    0H                                                         22
.MEND    MEND                                                             23
